function clsx(...parts: Array<string | false | null | undefined>) {
  return parts.filter(Boolean).join(" ");
}

function safeStr(v: unknown) {
  return typeof v === "string" ? v.trim() : "";
}

function upper(v: unknown) {
  return safeStr(v).toUpperCase();
}

function titleCase(v: unknown) {
  return safeStr(v)
    .toLowerCase()
    .split(/[\s_]+/)
    .filter(Boolean)
    .map((part) => part.charAt(0).toUpperCase() + part.slice(1))
    .join(" ");
}

function statusTone(status: string) {
  if (status === "IN_REVIEW") return "border-violet-400/30 bg-violet-400/12 text-violet-100";
  if (status === "COMPLETED") return "border-emerald-400/30 bg-emerald-400/12 text-emerald-100";
  if (status === "RELEASED") return "border-cyan-400/30 bg-cyan-400/12 text-cyan-100";
  if (status === "ASSIGNED") return "border-amber-400/30 bg-amber-400/12 text-amber-100";
  if (status === "AWAITING_ASSIGNMENT") {
    return "border-fuchsia-400/30 bg-fuchsia-400/12 text-fuchsia-100";
  }
  return "border-white/10 bg-white/5 text-white/65";
}

function priorityTone(priority: string) {
  if (priority === "URGENT") return "border-rose-400/30 bg-rose-400/12 text-rose-100";
  if (priority === "HIGH") return "border-amber-400/30 bg-amber-400/12 text-amber-100";
  if (priority === "NORMAL") return "border-cyan-400/30 bg-cyan-400/12 text-cyan-100";
  if (priority === "LOW") return "border-emerald-400/30 bg-emerald-400/12 text-emerald-100";
  return "border-white/10 bg-white/5 text-white/65";
}

function slaTone(state: string) {
  if (state === "overdue") return "border-rose-400/30 bg-rose-400/12 text-rose-100";
  if (state === "due_soon") return "border-amber-400/30 bg-amber-400/12 text-amber-100";
  if (state === "healthy") return "border-emerald-400/30 bg-emerald-400/12 text-emerald-100";
  return "border-white/10 bg-white/5 text-white/65";
}

function formatDateTime(value: Date | null | undefined) {
  if (!value) return "—";
  const d = value instanceof Date ? value : new Date(value);
  if (Number.isNaN(d.getTime())) return "—";

  return new Intl.DateTimeFormat("en-US", {
    month: "short",
    day: "numeric",
    year: "numeric",
    hour: "numeric",
    minute: "2-digit",
  }).format(d);
}

function queueSortScore(row: {
  slaState: string;
  assignee: string;
  status: string;
  dueAt: Date | null;
  priority: string;
  updatedAt: Date | null;
}) {
  const priorityWeight =
    row.priority === "URGENT"
      ? 0
      : row.priority === "HIGH"
        ? 1
        : row.priority === "NORMAL"
          ? 2
          : row.priority === "LOW"
            ? 3
            : 4;

  const slaWeight =
    row.slaState === "overdue"
      ? 0
      : row.slaState === "due_soon"
        ? 1
        : row.slaState === "healthy"
          ? 2
          : 3;

  const assignmentWeight = row.assignee ? 1 : 0;

  const statusWeight =
    row.status === "COMPLETED"
      ? 0
      : row.status === "IN_REVIEW"
        ? 1
        : row.status === "ASSIGNED"
          ? 2
          : row.status === "AWAITING_ASSIGNMENT"
            ? 3
            : row.status === "RELEASED"
              ? 4
              : 5;

  const dueWeight = row.dueAt ? row.dueAt.getTime() : Number.MAX_SAFE_INTEGER;
  const updatedWeight = row.updatedAt ? -row.updatedAt.getTime() : 0;

  return [slaWeight, priorityWeight, assignmentWeight, statusWeight, dueWeight, updatedWeight];
}

      await claimAndStart();
      return;
    }

    if (primaryAction === "start") {
      await sendAction("start");
      return;
    }

    if (primaryAction === "complete") {
      const saved = await saveOutcome("manual");
      if (!saved) {
        setError("Save review content before completing.");
        return;
      }

      const data = await sendAction("complete");
      if (data && FAST_MODE) {
        setReleaseDrawerOpen(true);
      }
      return;
    }

    if (primaryAction === "release") {
      setReleaseDrawerOpen(true);
    }
  }

      return;
    }

    const saved = await saveOutcome("manual");
    if (!saved) {
      setError("Save review content before releasing.");
      return;
    }

    const action = await sendAction("release");
    if (action) {
      setReleaseDrawerOpen(false);
      setMessage("Released to customer.");
      router.refresh();
    }
  }

function num(v: unknown) {
  const m = String(v ?? "").match(/\d+/);
  if (!m) return null;
  const n = Number(m[0]);
  return Number.isFinite(n) && n > 0 ? Math.floor(n) : null;
}

function upper(v: unknown) {
  return safeStr(v).toUpperCase();
}

function titleCase(value: unknown) {
  return safeStr(value)
    .toLowerCase()
    .split(/[\s_]+/)
    .filter(Boolean)
    .map((part) => part.charAt(0).toUpperCase() + part.slice(1))
    .join(" ");
}

function pickOne(v: string | string[] | undefined) {
  return Array.isArray(v) ? v[0] : v;
}

function boolish(v: string | string[] | undefined) {
  const s = upper(pickOne(v));
  return s === "1" || s === "TRUE" || s === "YES" || s === "OPEN";
}

function looksUnauthorized(err: any) {
  const code = String(err?.code || "");
  if (code === "UNAUTHORIZED") return true;
  const msg = String(err?.message || err || "").toLowerCase();
  return msg.includes("unauthorized") || msg.includes("forbidden");
}

function looksNoOrg(err: any) {
  const code = String(err?.code || "");
  if (
    code === "NO_ACTIVE_ORG" ||
    code === "ORG_REQUIRED" ||
    code === "NO_ORGANIZATION" ||
    code === "ORG_NOT_FOUND"
  ) {
    return true;
  }

  const msg = String(err?.message || err || "").toLowerCase();
  return (
    msg.includes("no active org") ||
    msg.includes("select org") ||
    msg.includes("no organization context") ||
    msg.includes("no organization") ||
    msg.includes("org required")
  );
}

function buildPath(
  base: string,
  params: Record<string, string | number | null | undefined>,
) {
  const sp = new URLSearchParams();
  for (const [k, v] of Object.entries(params)) {
    if (v == null) continue;
    const s = String(v).trim();
    if (!s) continue;
    sp.set(k, s);
  }
  const qs = sp.toString();
  return `${base}${qs ? `?${qs}` : ""}`;
}

function buildCurrentPath(runId: number, searchParams?: SearchParams) {
  return buildPath(`/review-desk/reviews/${runId}`, {
    releasePreview: boolish(searchParams?.releasePreview) ? 1 : undefined,
  });
}

function isPlainObject(v: unknown): v is Record<string, unknown> {
  return !!v && typeof v === "object" && !Array.isArray(v);
}

function clampScore(v: unknown): number | null {
  const n = typeof v === "number" ? v : Number(v);
  if (!Number.isFinite(n)) return null;
  return Math.max(0, Math.min(100, n));
}

function pickScorePreview(templateSnapshot: unknown): Record<string, unknown> | null {
  if (!isPlainObject(templateSnapshot)) return null;
  const sp = (templateSnapshot as any).scorePreview;
  return isPlainObject(sp) ? (sp as Record<string, unknown>) : null;
}

function pickSnapshotId(templateSnapshot: unknown): number | null {
  const sp = pickScorePreview(templateSnapshot);
  if (!sp) return null;
  return num(sp.snapshotId);
}

function pickSealedHash(templateSnapshot: unknown): string {
  const sp = pickScorePreview(templateSnapshot);
  if (!sp) return "";
  return safeStr(sp.sealedHash);
}

function pickSealedAt(templateSnapshot: unknown): string {
  const sp = pickScorePreview(templateSnapshot);
  if (!sp) return "";
  return safeStr(sp.sealedAt || sp.computedAt);
}

function pickOverallScore(templateSnapshot: unknown): number | null {
  const sp = pickScorePreview(templateSnapshot);
  if (sp) {
    const score = clampScore(sp.overall);
    if (score !== null) return score;
  }

  if (isPlainObject(templateSnapshot) && isPlainObject((templateSnapshot as any).scoring)) {
    return clampScore((templateSnapshot as any).scoring.overall);
  }

  return null;
}

function normalizedRunStatus(
  status: unknown,
  completedAt: unknown,
  templateSnapshot: unknown,
) {
  const sealed =
    !!pickSnapshotId(templateSnapshot) ||
    !!pickSealedHash(templateSnapshot) ||
    !!pickSealedAt(templateSnapshot);

  if (sealed || isCompletedRunStatus(status) || !!completedAt) return "COMPLETED";
  if (isReviewRunStatus(status)) return "REVIEW";
  if (isSubmittedRunStatus(status)) return "SUBMITTED";
  if (isDraftishRunStatus(status)) return "IN_PROGRESS";
  return upper(status) || "UNKNOWN";
}

function fmtDateTime(v?: Date | null | string) {
  if (!v) return "—";
  const dt = v instanceof Date ? v : new Date(String(v));
  if (!Number.isFinite(dt.getTime())) return "—";
  return new Intl.DateTimeFormat("en-US", {
    month: "short",
    day: "numeric",
    year: "numeric",
    hour: "numeric",
    minute: "2-digit",
  }).format(dt);
}

function progressionForStatus(
  status: unknown,
  packageCount: number,
  opts?: {
    reopened?: boolean;
    hasReleasedBaseline?: boolean;
    isCustomerTruvern?: boolean;
    truvernOpsReleased?: boolean;
    customerConfirmed?: boolean;
  },
) {
  const s = upper(status);
  const reopened = !!opts?.reopened;
  const hasReleasedBaseline = !!opts?.hasReleasedBaseline;
  const isCustomerTruvern = !!opts?.isCustomerTruvern;
  const truvernOpsReleased = !!opts?.truvernOpsReleased;
  const customerConfirmed = !!opts?.customerConfirmed;
  const packageSuffix =
    packageCount > 1 ? ` across ${packageCount} linked package assessments.` : ".";

  if (isCustomerTruvern && customerConfirmed) {
    return {
      currentStep: 4,
      heading: "Released customer baseline is active",
      body:
        "The Truvern expert release has already been confirmed by the customer. The released baseline is now active for this workspace.",
    };
  }

  if (isCustomerTruvern && truvernOpsReleased) {
    return {
      currentStep: 3,
      heading: "Customer confirm release is ready",
      body:
        "Truvern has published the expert release. Review the final preview and confirm release to make it the active customer-facing baseline.",
    };
  }

  if (isCustomerTruvern && !truvernOpsReleased) {
    return {
      currentStep: 3,
      heading: "Waiting for Truvern release",
      body:
        "Truvern has completed the expert review, but the expert release has not been published yet. Customer preview and confirm actions will unlock only after Truvern releases the review from the Truvern control center.",
    };
  }

  if (s === "ASSIGNED") {
    return {
      currentStep: 1,
      heading: reopened ? "Reopened review is queued to restart" : "Next step: start review",
      body: reopened
        ? "A previously released outcome exists, but a new review cycle has been opened. Starting review begins the next versioned cycle while preserving the last released result." +
          packageSuffix
        : "This assignment is ready to begin. Start review to move the workspace into active review handling" +
          packageSuffix,
    };
  }

  if (s === "IN_REVIEW") {
    return {
      currentStep: 2,
      heading: reopened ? "Reopened version is in progress" : "Next step: complete review",
      body: reopened
        ? "A prior released outcome remains preserved for downstream consumers while the reviewer works on a new draft version." +
          packageSuffix
        : "The review is actively being worked. Complete the review after the reviewer has finished analysis and findings are ready" +
          packageSuffix,
    };
  }

  if (s === "COMPLETED") {
    return {
      currentStep: 3,
      heading: reopened ? "Reopened version is ready to release" : "Next step: release review",
      body: reopened
        ? "The reopened review cycle is complete. Releasing now should publish the new version while superseding the previously released baseline downstream." +
          packageSuffix
        : "Review work is finished. Release review to finalize the lifecycle and mark the canonical package review as complete downstream" +
          packageSuffix,
    };
  }

  if (s === "RELEASED") {
    return {
      currentStep: 4,
      heading: hasReleasedBaseline ? "Released baseline is active" : "Lifecycle complete",
      body: hasReleasedBaseline
        ? "The latest released version is currently the active baseline."
        : "This review has already been released. No additional lifecycle action is available from this workspace.",
    };
  }

  return {
    currentStep: 0,
    heading: "Lifecycle unavailable",
    body: "No lifecycle progression is currently available for this review assignment.",
  };
}

function progressionStepTone(stepIndex: number, currentStep: number) {
  if (currentStep >= 4 && stepIndex <= 4) {
    return "border-emerald-400/25 bg-emerald-400/10 text-emerald-100";
  }
  if (stepIndex < currentStep) {
    return "border-emerald-400/25 bg-emerald-400/10 text-emerald-100";
  }
  if (stepIndex === currentStep) {
    return "border-cyan-400/30 bg-cyan-400/12 text-cyan-100";
  }
  return "border-white/10 bg-white/5 text-white/55";
}

function actionLabel(
  action: "start" | "complete" | "release" | "reopen" | null,
  packageCount: number,
  opts?: {
    reviewStatus?: string;
    hasReleasedBaseline?: boolean;
    isCustomerTruvern?: boolean;
    customerPreviewEnabled?: boolean;
    customerConfirmed?: boolean;
  },
) {
  const status = upper(opts?.reviewStatus);
  const hasReleasedBaseline = !!opts?.hasReleasedBaseline;
  const isCustomerTruvern = !!opts?.isCustomerTruvern;
  const customerPreviewEnabled = !!opts?.customerPreviewEnabled;
  const customerConfirmed = !!opts?.customerConfirmed;

  if (isCustomerTruvern && customerConfirmed) return "None";
  if (isCustomerTruvern && customerPreviewEnabled) {
    return "Review release preview → Confirm release";
  }
  if (isCustomerTruvern && !customerPreviewEnabled) {
    return "Waiting for Truvern release";
  }
  if (status === "RELEASED" && hasReleasedBaseline) {
    return packageCount > 1 ? "Reopen package review" : "Reopen review";
  }
  if (action === "reopen") return packageCount > 1 ? "Reopen package review" : "Reopen review";
  if (action === "start") return packageCount > 1 ? "Start package review" : "Start review";
  if (action === "complete") return packageCount > 1 ? "Complete package review" : "Complete review";
  if (action === "release") return packageCount > 1 ? "Release package review" : "Release review";
  return "None";
}

function latestTs(...values: Array<Date | null | undefined>) {
  let best: Date | null = null;
  for (const value of values) {
    if (!value) continue;
    if (!best || value.getTime() > best.getTime()) best = value;
  }
  return best;
}

function earliestTs(...values: Array<Date | null | undefined>) {
  let best: Date | null = null;
  for (const value of values) {
    if (!value) continue;
    if (!best || value.getTime() < best.getTime()) best = value;
  }
  return best;
}

function avgScore(values: Array<number | null | undefined>) {
  const valid = values.filter((v): v is number => typeof v === "number" && Number.isFinite(v));
  if (!valid.length) return null;
  return Math.round(valid.reduce((sum, v) => sum + v, 0) / valid.length);
}

function packageReviewLabel(count: number) {
  return count === 1 ? "Single assessment submission" : `Submitted package · ${count} assessments`;
}

function packageReleaseSummary(values: string[]) {
  const normalized = values.map((v) => upper(v)).filter(Boolean);
  if (!normalized.length) return "No saved outcomes";
  const releasedCount = normalized.filter((v) => isReleasedOutcomeState(v)).length;
  const internalCount = normalized.filter((v) => isInternalOnlyOutcomeState(v)).length;
  const draftCount = normalized.filter((v) => isDraftOutcomeState(v)).length;

  if (releasedCount === normalized.length) return "Fully released";
  if (releasedCount > 0) return `Partially released · ${releasedCount}/${normalized.length}`;
  if (internalCount === normalized.length) return "Internal only";
  if (draftCount === normalized.length) return "Draft only";
  return "Mixed release states";
}

function packageDecisionSummary(values: string[]) {
  const normalized = values.map((v) => upper(v)).filter(Boolean);
  if (!normalized.length) return "No saved decisions";
  const counts = new Map<string, number>();
  for (const value of normalized) counts.set(value, (counts.get(value) ?? 0) + 1);
  const sorted = [...counts.entries()].sort((a, b) => b[1] - a[1]);
  const [topLabel, topCount] = sorted[0];
  if (sorted.length === 1) return `${topLabel} across package`;
  return `${topLabel} leads · ${topCount}/${normalized.length}`;
}

function packageRiskSummary(values: string[]) {
  const normalized = values.map((v) => upper(v)).filter(Boolean);
  if (!normalized.length) return "No saved risk bands";
  if (normalized.includes("CRITICAL")) return "Contains critical residual risk";
  if (normalized.includes("HIGH") || normalized.includes("ELEVATED RISK")) {
    return "Contains elevated residual risk";
  }
  if (normalized.includes("MEDIUM") || normalized.includes("MODERATE RISK")) {
    return "Moderate residual risk across package";
  }
  if (normalized.every((v) => v === "LOW" || v === "LOW RISK")) {
    return "Low residual risk across package";
  }
  return "Mixed residual risk";
}

function packageRunStatusLabel(
  statuses: string[],
  releaseStates: string[],
  assignmentStatuses: string[],
) {
  const normalizedStatuses = statuses.map((s) => upper(s)).filter(Boolean);
  const normalizedReleases = releaseStates.map((s) => upper(s)).filter(Boolean);
  const normalizedAssignments = assignmentStatuses.map((s) => upper(s)).filter(Boolean);

  if (normalizedReleases.length > 0 && normalizedReleases.every((s) => isReleasedOutcomeState(s))) {
    return "RELEASED";
  }
  if (normalizedAssignments.some((s) => s === "COMPLETED")) return "COMPLETED";
  if (normalizedAssignments.some((s) => s === "IN_REVIEW")) return "IN_REVIEW";
  if (normalizedAssignments.some((s) => s === "ASSIGNED")) return "ASSIGNED";
  if (normalizedStatuses.length > 0 && normalizedStatuses.every((s) => s === "COMPLETED")) {
    return "COMPLETED";
  }
  if (normalizedStatuses.some((s) => s === "REVIEW")) return "REVIEW";
  if (normalizedStatuses.some((s) => s === "SUBMITTED")) return "SUBMITTED";
  if (normalizedStatuses.some((s) => s === "IN_PROGRESS")) return "IN_PROGRESS";
  return normalizedStatuses[0] || "UNKNOWN";
}

function choosePackagePrimaryRun<
  T extends { id: number; dueAt?: Date | null; submittedAt?: Date | null; createdAt?: Date | null },
>(runs: T[]) {
  if (!runs.length) return null;
  return [...runs].sort((a, b) => {
    const dueA = a.dueAt ? new Date(a.dueAt).getTime() : Number.MAX_SAFE_INTEGER;
    const dueB = b.dueAt ? new Date(b.dueAt).getTime() : Number.MAX_SAFE_INTEGER;
    if (dueA !== dueB) return dueA - dueB;

    const submittedA = a.submittedAt ? new Date(a.submittedAt).getTime() : Number.MAX_SAFE_INTEGER;
    const submittedB = b.submittedAt ? new Date(b.submittedAt).getTime() : Number.MAX_SAFE_INTEGER;
    if (submittedA !== submittedB) return submittedA - submittedB;

    const createdA = a.createdAt ? new Date(a.createdAt).getTime() : Number.MAX_SAFE_INTEGER;
    const createdB = b.createdAt ? new Date(b.createdAt).getTime() : Number.MAX_SAFE_INTEGER;
    if (createdA !== createdB) return createdA - createdB;

    return a.id - b.id;
  })[0];
}

function canActorOpenAssignedWorkspace(args: {
  actorUserId: string;
  actorRole: string;
  assignment:
    | {
        assignmentType?: string | null;
        status?: string | null;
        assessorUserId?: string | null;
        membership?: { userId?: string | null } | null;
      }
    | null
    | undefined;
}) {
  const assignmentType = upper(args.assignment?.assignmentType);
  const latestStatus = safeStr(args.assignment?.status).toUpperCase();
  const isLiveAssignedWorkspace =
    latestStatus === "ASSIGNED" ||
    latestStatus === "IN_REVIEW" ||
    latestStatus === "COMPLETED" ||
    latestStatus === "RELEASED";

  if (assignmentType === "TRUVERN" && isLiveAssignedWorkspace) return true;
  if (isInternalAssessorRole(args.actorRole)) return true;

  const assignedUserId =
    safeStr(args.assignment?.assessorUserId) ||
    safeStr(args.assignment?.membership?.userId) ||
    "";

  return !!assignedUserId && assignedUserId === safeStr(args.actorUserId) && isLiveAssignedWorkspace;
}

function normalizePreviewText(value: string) {
  return safeStr(value) || "—";
}

function buildReleaseDiffItems(args: {
  releasedExecutiveSummary: string;
  draftExecutiveSummary: string;
  releasedFindings: string;
  draftFindings: string;
  releasedRemediation: string;
  draftRemediation: string;
  releasedDecision: string;
  draftDecision: string;
  releasedRisk: string;
  draftRisk: string;
}): DiffItem[] {
  return [
    {
      label: "Executive summary",
      released: normalizePreviewText(args.releasedExecutiveSummary),
      draft: normalizePreviewText(args.draftExecutiveSummary),
      changed:
        normalizePreviewText(args.releasedExecutiveSummary) !==
        normalizePreviewText(args.draftExecutiveSummary),
    },
    {
      label: "Decision",
      released: normalizePreviewText(args.releasedDecision),
      draft: normalizePreviewText(args.draftDecision),
      changed:
        normalizePreviewText(args.releasedDecision) !==
        normalizePreviewText(args.draftDecision),
    },
    {
      label: "Residual risk",
      released: normalizePreviewText(args.releasedRisk),
      draft: normalizePreviewText(args.draftRisk),
      changed:
        normalizePreviewText(args.releasedRisk) !==
        normalizePreviewText(args.draftRisk),
    },
    {
      label: "Key findings",
      released: normalizePreviewText(args.releasedFindings),
      draft: normalizePreviewText(args.draftFindings),
      changed:
        normalizePreviewText(args.releasedFindings) !==
        normalizePreviewText(args.draftFindings),
    },
    {
      label: "Remediation / follow-up",
      released: normalizePreviewText(args.releasedRemediation),
      draft: normalizePreviewText(args.draftRemediation),
      changed:
        normalizePreviewText(args.releasedRemediation) !==
        normalizePreviewText(args.draftRemediation),
    },
  ];
}

function diffPillTone(label: string, value: string) {
  const key = upper(label);
  if (key === "DECISION") return decisionPillTone(value);
  if (key === "RESIDUAL RISK") return riskPillTone(value);
  return "border-white/10 bg-white/5 text-white/75";
}

function diffDisplayValue(label: string, value: string) {
  const key = upper(label);
  if (key === "DECISION") return prettyDecisionLabel(value);
  if (key === "RESIDUAL RISK") return prettyRiskLabel(value);
  return value;
}

export default clerkMiddleware(async (authFn, req) => {
  const res = NextResponse.next();

  // Always hydrate Clerk request auth
  const a = await authFn();

  if (process.env.NODE_ENV !== "production") {
    res.headers.set("x-truvern-mw", "1");
  }

  // Public routes always allowed
  if (isPublicRoute(req)) return res;

  // API routes: never redirect from middleware
  if (isApiNoRedirectRoute(req)) return res;

  // Protected routes: sign-in required only
  if (!a.userId) {
    return a.redirectToSignIn({ returnBackUrl: req.url });
  }

  /**
   * IMPORTANT:
   * Do NOT perform organization gating in middleware.
   * Let server pages/routes resolve org context via requireDbOrganization().
   * This prevents select-org/onboarding loops and keeps org switching invisible.
   */

  return res;
});

// lib/truvern-ops-access.ts
import { auth } from "@clerk/nextjs/server";

function safeStr(v: unknown) {
  return typeof v === "string" ? v.trim() : "";
}

export function parseAllowedUsers() {
  return safeStr(process.env.TRUVERN_OPS_USERS)
    .split(",")
    .map((x) => x.trim())
    .filter(Boolean);
}

export async function requireTruvernOperator() {
  const { userId } = await auth();

  if (!userId) {
    throw new Error("Unauthorized");
  }

  const allowed = parseAllowedUsers();

  if (!allowed.includes(userId)) {
    throw new Error("Not authorized for Truvern ops.");
  }

  return { userId };
}

export async function isTruvernOperator() {
  try {
    const { userId } = await auth();

    if (!userId) return false;

    const allowed = parseAllowedUsers();
    return allowed.includes(userId);
  } catch {
    return false;
  }
}

export const ORG_ROLE_OPTIONS = [
  "OWNER",
  "ADMIN",
  "ANALYST",
  "VIEWER",
  "VENDOR",
  "MEMBER",
] as const;

export type OrgRoleName = (typeof ORG_ROLE_OPTIONS)[number];

export type ReviewDeskLifecycleAction =
  | "start"
  | "complete"
  | "release"
  | "reopen"
  | "save_findings";

export type ReviewDeskRoleName = Exclude<OrgRoleName, "VENDOR">;

export function safeStr(v: unknown) {
  return typeof v === "string" ? v.trim() : "";
}

export function normalizeOrgRole(role: unknown) {
  return safeStr(role).toUpperCase();
}

export function isOrgRole(role: unknown): role is OrgRoleName {
  return (ORG_ROLE_OPTIONS as readonly string[]).includes(normalizeOrgRole(role));
}

/** Full org control */
export function isManageRole(role: unknown) {
  const r = normalizeOrgRole(role);
  return r === "OWNER" || r === "ADMIN";
}

/** Can actively review (write actions) */
export function isInternalAssessorRole(role: unknown) {
  const r = normalizeOrgRole(role);
  return r === "OWNER" || r === "ADMIN" || r === "ANALYST";
}

/** Can at least SEE review desk (read access) */
export function isReviewDeskViewerRole(role: unknown) {
  const r = normalizeOrgRole(role);
  return (
    r === "OWNER" ||
    r === "ADMIN" ||
    r === "ANALYST" ||
    r === "VIEWER" ||
    r === "MEMBER"
  );
}

/** Main gate for review desk access */
export function isReviewDeskRole(role: unknown) {
  return isReviewDeskViewerRole(role);
}

/** Back-compat alias: any role that can actively write within Review Desk */
export function canPerformReviewActions(role: unknown) {
  return isInternalAssessorRole(role);
}

/** Normalize role specifically for Review Desk RBAC */
export function normalizeReviewDeskRole(role: unknown) {
  return normalizeOrgRole(role);
}

/** Explicit read-access helper for Review Desk pages */
export function canAccessReviewDesk(role: unknown) {
  return isReviewDeskViewerRole(role);
}

/** Findings can be saved by active reviewer roles */
export function canSaveReviewFindings(role: unknown) {
  const r = normalizeReviewDeskRole(role);
  return r === "OWNER" || r === "ADMIN" || r === "ANALYST";
}

/** Only control roles can release a review */
export function canReleaseReview(role: unknown) {
  const r = normalizeReviewDeskRole(role);
  return r === "OWNER" || r === "ADMIN";
}

/** Only control roles can reopen a released review */
export function canReopenReview(role: unknown) {
  const r = normalizeReviewDeskRole(role);
  return r === "OWNER" || r === "ADMIN";
}

/** Canonical lifecycle permission helper shared by client + server */
export function canPerformReviewLifecycleAction(
  role: unknown,
  action: ReviewDeskLifecycleAction | string | null | undefined,
) {
  const r = normalizeReviewDeskRole(role);
  const a = safeStr(action).toLowerCase();

  if (!r || !a) return false;

  if (r === "OWNER" || r === "ADMIN") {
    return (
      a === "start" ||
      a === "complete" ||
      a === "release" ||
      a === "reopen" ||
      a === "save_findings"
    );
  }

  if (r === "ANALYST") {
    return a === "start" || a === "complete" || a === "save_findings";
  }

  return false;
}
