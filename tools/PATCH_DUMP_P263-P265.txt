function clsx(...parts: Array<string | false | null | undefined>) {
  return parts.filter(Boolean).join(" ");
}

function safeStr(v: any) {
  return typeof v === "string" ? v.trim() : "";
}

function tierRank(t: PlanTier) {
  return t === "ENTERPRISE" ? 3 : t === "PRO" ? 2 : 1;
}

function tierPill(t?: PlanTier | null) {
  const v = (safeStr(t) || "FREE").toUpperCase() as PlanTier;
  if (v === "ENTERPRISE") return "border-fuchsia-400/20 bg-fuchsia-400/10 text-fuchsia-200";
  if (v === "PRO") return "border-cyan-400/20 bg-cyan-400/10 text-cyan-200";
  return "border-emerald-400/20 bg-emerald-400/10 text-emerald-200";
}

function reasonLabelFromType(t?: string) {
  if (t === "question_added") return "Question added";
  if (t === "question_removed") return "Question removed";
  if (t === "question_changed") return "Question changed";
  if (t === "section_added") return "Section added";
  if (t === "section_removed") return "Section removed";
  return "Change";
}

function getDevOrgId(): number | null {
  if (typeof window === "undefined") return null;
  const url = new URL(window.location.href);
  const raw =
    url.searchParams.get("devOrgId") ||
    url.searchParams.get("orgId") ||
    window.localStorage.getItem("truvern_dev_orgid") ||
    "";
  const n = Number(String(raw).trim());
  if (Number.isFinite(n) && n > 0) {
    try {
      window.localStorage.setItem("truvern_dev_orgid", String(n));
    } catch {}
    return n;
  }
  return null;
}

function buildDevHeaders(devOrgId: number | null) {
  const h: Record<string, string> = { "x-dev-bypass": "1" };
  if (devOrgId) h["x-dev-orgid"] = String(devOrgId);
  return h;
}

type RiskLevel = "LOW" | "MODERATE" | "HIGH" | "CRITICAL";
function normalizeRiskLevel(v: any): RiskLevel | null {
  const s = String(v || "").trim().toUpperCase();
  if (!s) return null;
  if (s === "LOW" || s === "L") return "LOW";
  if (s === "MODERATE" || s === "MEDIUM" || s === "M") return "MODERATE";
  if (s === "HIGH" || s === "H") return "HIGH";

}

function isClosedStatus(status: any) {
  const s = String(status || "").trim().toUpperCase();
  return s === "CLOSED" || s === "DONE" || s === "COMPLETED" || s === "FULFILLED" || s === "ARCHIVED";

function computeDerivedRisk(args: {
  openIssues: any[];
  openEvidenceRequests: any[];
  overdueEvidenceRequests: any[];
}): { level: RiskLevel; score: number; reason: string } {
  const { openIssues, openEvidenceRequests, overdueEvidenceRequests } = args;

  const sevWeight = (sev: any) => {
    const s = String(sev || "").trim().toUpperCase();
    if (s === "CRITICAL") return 20;
    if (s === "HIGH") return 12;
    if (s === "MEDIUM" || s === "MODERATE") return 7;
    if (s === "LOW") return 3;
    return 5;
  };

  const issueScore = openIssues.reduce((acc, it) => acc + sevWeight((it as any).severity), 0);
  const overdueScore = overdueEvidenceRequests.length * 6;
  const openReqScore = Math.max(0, openEvidenceRequests.length - overdueEvidenceRequests.length) * 2;

  const score = issueScore + overdueScore + openReqScore;

  let level: RiskLevel = "LOW";
  if (score >= 45) level = "CRITICAL";
  else if (score >= 25) level = "HIGH";
  else if (score >= 12) level = "MODERATE";

  const reason =
    overdueEvidenceRequests.length > 0
      ? `Overdue evidence (${overdueEvidenceRequests.length}) + open issues (${openIssues.length})`
      : `Open issues (${openIssues.length}) + open requests (${openEvidenceRequests.length})`;

  return { level, score, reason };
}

function badgeToneForFreshness(days: number | null): { label: string; cls: string } {
  if (days == null) {
    return { label: "Unknown", cls: "border-white/10 bg-white/5 text-slate-200" };
  }
  if (days <= 90) return { label: "Fresh", cls: "border-emerald-300/25 bg-emerald-400/10 text-emerald-100" };
  if (days <= 180) return { label: "Aging", cls: "border-amber-300/25 bg-amber-400/10 text-amber-100" };
  return { label: "Stale", cls: "border-rose-300/25 bg-rose-400/10 text-rose-100" };
}

type ValidityStatus = "VALID" | "EXPIRING" | "EXPIRED" | "UNKNOWN";
function computeValidity(
  validUntil: Date | null,
  now: Date
): { status: ValidityStatus; label: string; cls: string; sub: string } {
  if (!validUntil) {
    return {
      status: "UNKNOWN",
      label: "No expiry",
      cls: "border-white/10 bg-white/5 text-slate-200",
      sub: "No validUntil on latest evidence",
    };
  }

  const du = validUntil instanceof Date ? validUntil : new Date(validUntil);
  if (!Number.isFinite(du.getTime())) {
    return {
      status: "UNKNOWN",
      label: "No expiry",
      cls: "border-white/10 bg-white/5 text-slate-200",
      sub: "validUntil is invalid",
    };
  }

  const d = daysUntil(du, now);

  if (d < 0) {
    const expiredAgo = Math.abs(Math.floor((now.getTime() - du.getTime()) / (1000 * 60 * 60 * 24)));
    return {
      status: "EXPIRED",
      label: "Expired",
      cls: "border-rose-300/25 bg-rose-400/10 text-rose-100",
      sub: `Expired ${expiredAgo}d ago`,
    };
  }

  if (d <= 30) {
    return {
      status: "EXPIRING",
      label: "Expiring",
      cls: "border-amber-300/25 bg-amber-400/10 text-amber-100",
      sub: `Expires in ${d}d`,
    };
  }

  return {
    status: "VALID",
    label: "Valid",
    cls: "border-emerald-300/25 bg-emerald-400/10 text-emerald-100",
    sub: `Expires in ${d}d`,
  };
}

}

type CompareSummary = {
  sectionsAdded: number;
  sectionsRemoved: number;
  sectionsChanged: number;

type TemplateSummary = {
  id: number;
  name: string;

type QuestionKind = "YES_NO" | "TEXT" | "SELECT" | "MULTI_SELECT" | "NUMBER";

type UIQuestion = {
  _key: string;
  text: string;
  kind: QuestionKind;
  required: boolean;
  optionsText?: string;
  orderIndex: number;
};

type UISection = {
  _key: string;
  title: string;
  description?: string;
  order: number;
  collapsed?: boolean;
  questions: UIQuestion[];
};

type TemplateStructure = {
  templateId: number;
  sections?: UISection[];
  source?: "published" | "draft";
  draftUpdatedAt?: string | null;
};

function clsx(...parts: Array<string | false | null | undefined>) {
  return parts.filter(Boolean).join(" ");
}

function fmtDate(d?: string | null) {
  if (!d) return "—";
  const dt = new Date(d);
  if (Number.isNaN(dt.getTime())) return "—";
  return dt.toLocaleDateString(undefined, { year: "numeric", month: "short", day: "2-digit" });
}

function asArray<T>(v: any): T[] {
  return Array.isArray(v) ? (v as T[]) : [];
}

function safeStr(v: any) {
  return typeof v === "string" ? v.trim() : "";
}

function pickFirstString(...vals: any[]) {
  for (const v of vals) {
    const s = safeStr(v);
    if (s) return s;
  }
  return "";
}

function kindLabel(kind?: string) {
  const k = safeStr(kind) || "TEXT";
  return k.replace(/_/g, " ");
}

function normalizeStructure(raw: any): TemplateStructure | null {
  if (!raw || typeof raw !== "object") return null;

  const templateIdRaw = (raw as any).templateId;
  const templateId = Number(templateIdRaw);

  const rawSections = asArray<any>((raw as any).sections);
  if (!Number.isFinite(templateId) || templateId <= 0) {
    if (!rawSections.length) return null;
  }

  const sections: UISection[] = rawSections.map((s: any, i: number) => {
    const secKey = pickFirstString(s?._key, s?.key, s?.id, `s-${i}`);
    const secTitle = pickFirstString(s?.title, s?.name, s?.label, `Section ${i + 1}`);
    const secDesc = pickFirstString(s?.description, s?.helpText, "");
    const secOrder = Number.isFinite(Number(s?.order)) ? Number(s.order) : i;
    const secCollapsed = typeof s?.collapsed === "boolean" ? s.collapsed : false;

    const rawQs = asArray<any>(s?.questions);

    const questions: UIQuestion[] = rawQs.map((q: any, qi: number) => {
      const qKey = pickFirstString(q?._key, q?.key, q?.id, `${secKey}-q-${qi}`);

      const qText = pickFirstString(q?.text, q?.prompt, q?.question, q?.label, "");

      const qKindRaw = pickFirstString(q?.kind, q?.type, "TEXT").toUpperCase();
      const qKind: QuestionKind = (["YES_NO", "TEXT", "SELECT", "MULTI_SELECT", "NUMBER"].includes(qKindRaw)
        ? qKindRaw
        : "TEXT") as QuestionKind;

      const required = !!q?.required;
      const optionsText = pickFirstString(q?.optionsText, q?.options, q?.choices, "");
      const orderIndex = Number.isFinite(Number(q?.orderIndex))
        ? Number(q.orderIndex)
        : Number.isFinite(Number(q?.order))
        ? Number(q.order)
        : qi;

      return {
        _key: String(qKey),
        text: qText || "(Untitled question)",
        kind: qKind,
        required,
        optionsText,
        orderIndex,
      };
    });

    return {
      _key: String(secKey),
      title: secTitle || `Section ${i + 1}`,
      description: secDesc || "",
      order: secOrder,
      collapsed: secCollapsed,
      questions,
    };
  });

  return {
    templateId: Number.isFinite(templateId) ? templateId : 0,
    sections,
    source: (raw as any).source,
    draftUpdatedAt: (raw as any).draftUpdatedAt ?? null,
  };
}

function getDevOrgIdFromUrl(): number | null {
  if (typeof window === "undefined") return null;

function getDevOrgIdFromStorage(): number | null {
  if (typeof window === "undefined") return null;
  const raw = window.localStorage.getItem("truvern_dev_orgid") || "";
  const n = Number(String(raw).trim());
  return Number.isFinite(n) && n > 0 ? n : null;
}

function setDevOrgIdInStorage(n: number) {
  if (typeof window === "undefined") return;
  try {
    window.localStorage.setItem("truvern_dev_orgid", String(n));
  } catch {}
}

    const installedStructure = draftCol ? await readJsonColumnById(draftCol, installed.id) : null;
    const installedCatalogVersion = await readCatalogVersionById(cols, installed.id);

    const diff = diffPresetVsInstalled(preset, installedStructure);
    const stats = computePresetStats(preset);

    const presetVersion = preset.version ?? null;
    const upgradeNeeded = compareVersion(installedCatalogVersion, presetVersion) < 0;

    return NextResponse.json({
      ok: true,
      preset: { key: preset.key, version: presetVersion, stats },
      installed: { id: installed.id, version: (installed as any).version ?? null, catalogVersion: installedCatalogVersion },
      meta: { draftColumn: draftCol, upgradeNeeded },
      diff,
    });
  } catch (e: any) {
    const msg = safeStr(e?.message) || "Server error";
    const code = safeStr(e?.code);
    return NextResponse.json({ ok: false, error: msg, code }, { status: 500 });

export async function POST(req: Request) {
  try {
    const org = await requireDbOrganization();
    const body = await req.json().catch(() => ({}));
    const key = safeStr((body as any)?.key);

    if (!key) {
      return NextResponse.json({ ok: false, error: "Missing key" }, { status: 400 });
    }

    const preset = getPresetByKey(key);
    if (!preset) {
      return NextResponse.json({ ok: false, error: "Unknown preset key" }, { status: 404 });
    }

    // Tier gating
    const orgTier = await getOrgPlanTier(org.id);
    const requires = ((preset as any).requiresTier || "FREE") as PlanTier;
    if (tierRank(orgTier) < tierRank(requires)) {
      return NextResponse.json(
        { ok: false, error: `Requires ${requires}`, required: requires, current: orgTier },
        { status: 402 }
      );
    }

    const cols = await getAssessmentTemplateColumns();
    const hasCatalogKey = cols.has("catalogKey");

    // Find installed (catalogKey if present, else by name)
    const installed = await prisma.assessmentTemplate.findFirst({
      where: {
        organizationId: org.id,
        ...(hasCatalogKey ? ({ catalogKey: key } as any) : { name: preset.name }),
      } as any,
      select: { id: true, version: true } as any,
    });

    if (!installed) {
      return NextResponse.json({ ok: false, error: "Not installed" }, { status: 404 });
    }

    const draftCol = pickFirstExisting(cols, [
      "draft",
      "draftJson",
      "draftStructure",
      "draftContent",
      "structure",
      "content",
    ]);

    if (!draftCol) {
      return NextResponse.json(
        { ok: false, error: "No draft/structure column found to upgrade into." },
        { status: 500 }
      );
    }

    const currentCatalogVersion = await readCatalogVersionById(cols, installed.id);
    const toVersion = preset.version ?? null;

    // Only upgrade if newer
    const needsUpgrade = compareVersion(currentCatalogVersion, toVersion) < 0;

    if (!needsUpgrade) {

        status: "no_change",
        templateId: installed.id,
        toVersion,
        draftColumn: draftCol,
        catalogVersion: currentCatalogVersion,
      });

    if (cols.has("catalogVersion")) {
      await setScalarColumnById("catalogVersion", installed.id, toVersion);
    }

    return NextResponse.json({
      ok: true,
      status: "upgraded_draft",
      templateId: installed.id,
      toVersion,
      draftColumn: draftCol,
    });
  } catch (e: any) {
    const msg = safeStr(e?.message) || "Server error";
    const code = safeStr(e?.code);
    return NextResponse.json({ ok: false, error: msg, code }, { status: 500 });
  }
}

export function compareVersion(a?: string | null, b?: string | null): number {
  // returns: -1 if a<b, 0 if equal, 1 if a>b
  const pa = safeStr(a).split(".").filter(Boolean).map((x) => Number(x));
  const pb = safeStr(b).split(".").filter(Boolean).map((x) => Number(x));
  const len = Math.max(pa.length, pb.length);
  for (let i = 0; i < len; i++) {
    const na = Number.isFinite(pa[i]) ? pa[i] : 0;
    const nb = Number.isFinite(pb[i]) ? pb[i] : 0;
    if (na < nb) return -1;
    if (na > nb) return 1;
  }
  return 0;
}

export function computePresetStats(preset: CatalogPreset): PresetStats {
  let questions = 0;
  let required = 0;
  for (const s of preset.structure.sections) {
    for (const q of s.questions) {

      if (q.required) required++;
    }

function indexByTitle(struct: any): Record<string, CatalogSection> {
  const out: Record<string, CatalogSection> = {};
  const sections = Array.isArray(struct?.sections) ? struct.sections : [];
  for (const s of sections) {
    const title = safeStr(s?.title);
    if (!title) continue;
    out[title] = {
      title,
      questions: Array.isArray(s?.questions)
        ? s.questions.map((q: any) => ({
            text: safeStr(q?.text),
            kind: safeStr(q?.kind),
            required: !!q?.required,
          }))
        : [],
    };
  }
  return out;
}

function questionKey(q: CatalogQuestion) {
  return safeStr(q.text).toLowerCase();
}

export function diffPresetVsInstalled(preset: CatalogPreset, installedStructure: any): DiffResult {
  const changes: DiffResult["changes"] = [];

  const pIdx = indexByTitle(preset.structure);
  const iIdx = indexByTitle(installedStructure);

  const pTitles = Object.keys(pIdx);
  const iTitles = Object.keys(iIdx);

  const sectionsAdded = pTitles.filter((t) => !iIdx[t]);
  const sectionsRemoved = iTitles.filter((t) => !pIdx[t]);
  const sectionsChanged: string[] = [];

  let questionsAdded = 0;
  let questionsRemoved = 0;
  let questionsChanged = 0;

  for (const t of sectionsAdded) changes.push({ section: t, type: "section_added" });
  for (const t of sectionsRemoved) changes.push({ section: t, type: "section_removed" });

  for (const t of pTitles) {
    if (!iIdx[t]) continue;
    const ps = pIdx[t];
    const is = iIdx[t];

    const pQ = new Map(ps.questions.map((q) => [questionKey(q), q]));
    const iQ = new Map(is.questions.map((q) => [questionKey(q), q]));

    for (const [k, q] of pQ.entries()) {
      if (!iQ.has(k)) {
        questionsAdded++;
        changes.push({ section: t, type: "question_added", questionText: q.text, to: q });
      }
    }
    for (const [k, q] of iQ.entries()) {
      if (!pQ.has(k)) {
        questionsRemoved++;
        changes.push({ section: t, type: "question_removed", questionText: q.text, from: q });
      }
    }

    for (const [k, pq] of pQ.entries()) {
      const iq = iQ.get(k);
      if (!iq) continue;

      const kindChanged = safeStr(pq.kind) !== safeStr(iq.kind);
      const reqChanged = !!pq.required !== !!iq.required;
      if (kindChanged || reqChanged) {
        questionsChanged++;
        changes.push({
          section: t,
          type: "question_changed",
          questionText: pq.text,
          from: { kind: iq.kind, required: iq.required },
          to: { kind: pq.kind, required: pq.required },
        });
      }
    }

    if (changes.some((c) => c.section === t && c.type.startsWith("question_"))) {
      if (!sectionsChanged.includes(t)) sectionsChanged.push(t);
    }
  }

  return {
    sectionsAdded,
    sectionsRemoved,
    sectionsChanged,
    questionsAdded,
    questionsRemoved,
    questionsChanged,
    changes,
  };
}

export function recommendPresets(profile: VendorProfile) {
  const p = profile || {};
  const out: Recommendation[] = [];

  // Candidate set (keep all for now)
  const candidates = PRESETS.slice();

  for (const preset of candidates) {
    const r = baseRec(preset);

    /* ------------------------- baseline fallback -------------------------- */
    if (preset.key === "vendor_security_baseline_v1") {
      r.score += 25;
      r.reasons.push("Great starter when you want a fast baseline for low/medium risk vendors.");
      if (p.criticality === "HIGH") {
        r.score += 10;
        r.reasons.push("High criticality still benefits from a quick baseline while you pick deeper frameworks.");
      }
    }

    /* ----------------------------- SOC 2 -------------------------------- */
    if (preset.framework === "SOC 2") {
      if (p.isCloudSaaS) {
        r.score += 25;
        r.reasons.push("SOC 2 is a strong default for SaaS vendors handling business data.");
      }
      if (p.storesCustomerPII || p.dataSensitivity === "CONFIDENTIAL" || p.dataSensitivity === "RESTRICTED") {
        r.score += 15;
        r.reasons.push("Vendor handles sensitive data; SOC 2 coverage is a good fit.");
      }
      if (p.criticality === "HIGH") {
        r.score += 10;
        r.reasons.push("High criticality vendor; SOC 2 is board-familiar and evidence-oriented.");
      }
      // IMPORTANT: if payments, SOC2 is still relevant, but PCI should lead
      if (p.processesPayments) {
        r.score += 3;
        r.reasons.push("SOC 2 can complement payment controls, but PCI should be primary when card data is involved.");
      }
    }

    /* --------------------------- ISO 27001 ------------------------------- */
    if (preset.framework === "ISO 27001") {
      if (p.companySize === "ENTERPRISE" || p.criticality === "HIGH") {
        r.score += 25;
        r.reasons.push("ISO 27001 aligns well to enterprise procurement and ISMS governance.");
      }
      if (p.isInfrastructureProvider) {
        r.score += 10;
        r.reasons.push("Infrastructure vendors benefit from ISMS-aligned control validation.");
      }
    }

    /* ---------------------------- NIST CSF ------------------------------- */
    if (preset.framework === "NIST CSF") {
      if (p.criticality === "HIGH") {
        r.score += 30;
        r.reasons.push("NIST CSF is ideal for high-criticality vendors (Detect/Respond/Recover depth).");
      }
      if (p.isInfrastructureProvider) {
        r.score += 20;
        r.reasons.push("Infrastructure providers map well to NIST CSF operational resilience controls.");
      }
    }

    /* ----------------------------- PCI DSS ------------------------------- */
    if (preset.framework === "PCI DSS") {
      if (p.processesPayments) {
        // Make PCI decisively the top pick for payment processors
        r.score += 85;
        r.reasons.push("Vendor processes payments; PCI DSS is the correct starting framework.");
        if (p.criticality === "HIGH") {
          r.score += 10;
          r.reasons.push("High criticality payments vendor: prioritize PCI-aligned evidence and control validation.");
        }
        if (p.storesCustomerPII) {
          r.score += 5;
          r.reasons.push("PII + payments raises risk; PCI-focused diligence reduces audit surprises.");
        }
      } else {
        r.score -= 15;
      }
    }

    /* ------------------------------ HIPAA -------------------------------- */
    if (preset.framework === "HIPAA") {
      if (p.isHealthcareVendor) {
        r.score += 80;
        r.reasons.push("Healthcare/PHI context; HIPAA safeguards are the correct baseline.");
      } else {
        r.score -= 15;
      }
    }

    /* ------------------------- enterprise pack --------------------------- */
    if (preset.key.includes("enterprise_pack")) {
      if (p.criticality === "HIGH" || p.companySize === "ENTERPRISE") {
        r.score += 15;
        r.reasons.push("Enterprise pack adds deeper evidence prompts and board-ready output for high-risk vendors.");
      } else {
        r.score -= 30;
      }
    }

    /* -------------------------- general modifiers ------------------------ */
    if (p.hasSubprocessors) r.score += 5;
    if (p.region === "EU" || p.region === "GLOBAL") r.score += 3;

    out.push(r);
  }

  out.sort((a, b) => b.score - a.score);
  return out.slice(0, 8);
}

export function tierRank(t: PlanTier) {
  if (t === "ENTERPRISE") return 3;
  if (t === "PRO") return 2;
  return 1;
}

function safeStr(v: unknown) {
  return typeof v === "string" ? v.trim() : "";
}

async function inferPlanTierFromDb(orgId: number): Promise<PlanTier> {
  // Best effort: adapt to your schema without breaking builds.
  // If you have Organization.planTier, we’ll use it.

    });

    // @ts-expect-error

      // @ts-expect-error

    });

    // @ts-expect-error

    if (pt === "ENTERPRISE" || pt === "PRO" || pt === "FREE") return pt as PlanTier;
  } catch {
    // ignore
  }

  return "FREE";
}

async function headerPlanOverride(): Promise<PlanTier | null> {
  // Useful for dev scripts + easy toggling without DB changes:
  // x-dev-plan-tier: FREE|PRO|ENTERPRISE
  try {
    const h = await headers();
    const raw = safeStr(h.get("x-dev-plan-tier"));
    const v = raw.toUpperCase();
    if (v === "FREE" || v === "PRO" || v === "ENTERPRISE") return v as PlanTier;
  } catch {
    // ignore
  }
  return null;
}

  // ✅ IMPORTANT: allow explicit header override to win (even if BYPASS_PLAN_GATING=1)
  const override = await headerPlanOverride();
  if (override) return override;

  // Bypass becomes the “default” tier only when no explicit override is provided
  if (bypass()) return "ENTERPRISE";

      current: tier,

  }

  return { tier };
}

// lib/org-db.ts
import prisma from "@/lib/prisma";
import { headers } from "next/headers";
import { auth } from "@clerk/nextjs/server";

function safeStr(v: unknown) {
  return typeof v === "string" ? v.trim() : "";
}
function parseIntSafe(v: unknown): number | null {
  const n = Number(String(v ?? "").trim());
  return Number.isFinite(n) && n > 0 ? Math.floor(n) : null;
}

export class OrgResolutionError extends Error {
  code: string;
  details?: Record<string, any>;

  constructor(code: string, message: string, details?: Record<string, any>) {
    super(message);
    this.name = "OrgResolutionError";
    this.code = code;
    this.details = details;
  }
}

async function getHeader(name: string) {
  const h = await headers();
  return safeStr(h.get(name));
}

async function readDevBypass() {
  const v = (await getHeader("x-dev-bypass")).toLowerCase();
  return v === "1" || v === "true" || v === "yes";
}

async function readDevOrgIdHeader(): Promise<number | null> {
  const v = await getHeader("x-dev-orgid");
  return parseIntSafe(v);
}

function readDevOrgIdEnv(): number | null {
  // ✅ Use any of these in local dev to make browser navigation work:
  // DEV_ORG_ID=2
  // TRUVERN_DEV_ORG_ID=2
  const v =
    safeStr(process.env.DEV_ORG_ID) || safeStr(process.env.TRUVERN_DEV_ORG_ID);
  return parseIntSafe(v);
}

async function resolveOrgById(orgId: number, route?: string) {
  const org = await prisma.organization.findUnique({
    where: { id: orgId },
  });

  if (!org) {
    throw new OrgResolutionError("ORG_NOT_FOUND", `Organization ${orgId} not found.`, {
      route,
      orgId,
    });
  }

  return org as any;
}

async function resolveOrgByClerkOrgId(clerkOrgId: string, route?: string) {
  // If your schema uses a different field than clerkOrgId, adjust this where clause.
  const org = await prisma.organization.findFirst({
    where: { clerkOrgId },
  });

  if (!org) {
    throw new OrgResolutionError(
      "ORG_NOT_FOUND",
      "No database organization found for this Clerk organization.",
      { route, clerkOrgId }
    );
  }

  return org as any;
}

/**
 * requireDbOrganization()
 *
 * Priority:
 *  1) Dev bypass headers: x-dev-bypass + x-dev-orgid
 *  2) Clerk org context: auth().orgId -> organization.clerkOrgId
 *  3) DEV ONLY fallback: DEV_ORG_ID / TRUVERN_DEV_ORG_ID (browser-friendly local dev)
 */
export async function requireDbOrganization(route?: string) {
  // 1) Dev bypass headers (curl/scripts)
  const bypass = await readDevBypass();
  if (bypass) {
    const devOrgId = await readDevOrgIdHeader();
    if (!devOrgId) {
      throw new OrgResolutionError(
        "ORG_CONTEXT_MISSING",
        "Dev bypass is enabled but x-dev-orgid is missing.",
        { route }
      );
    }
    return await resolveOrgById(devOrgId, route);
  }

  // 2) Clerk org context
  const a = auth();
  const clerkOrgId = safeStr((a as any)?.orgId);
  if (clerkOrgId) {
    return await resolveOrgByClerkOrgId(clerkOrgId, route);
  }

  // 3) DEV fallback (browser navigation in local dev)
  if (process.env.NODE_ENV !== "production") {
    const envOrgId = readDevOrgIdEnv();
    if (envOrgId) {
      return await resolveOrgById(envOrgId, route);
    }
  }

  throw new OrgResolutionError(
    "ORG_CONTEXT_MISSING",
    "No organization context found for this request.",
    { route }
  );
}

// Optional helper (non-throwing)
export async function getDbOrganizationMaybe(route?: string) {
  try {
    return await requireDbOrganization(route);
  } catch {
    return null;
  }
}
